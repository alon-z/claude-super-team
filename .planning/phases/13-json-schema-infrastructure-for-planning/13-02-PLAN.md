   ```bash
   echo "=== ROADMAP_TRIMMED ==="
   if [ -z "${PHASE_NUM:-}" ]; then
     echo "(no phase specified -- use full roadmap)"
   elif [ ! -f "$ROADMAP_FILE" ]; then
     echo "(missing)"
   else
     echo "--- Phases Overview ---"
     if [ "$_JQ_AVAILABLE" = "true" ] && [ -f "$P/ROADMAP.json" ]; then
       jq -r '.phases[] | "- [\(if .complete then "x" else " " end)] **Phase \(.id): \(.name)**"' "$P/ROADMAP.json" 2>/dev/null
     else
       # existing awk block for phases list
       awk '
         /^## Phases/ { found=1; next }
         found && /^## / { exit }
         found { print }
       ' "$ROADMAP_FILE" 2>/dev/null
     fi

     # Phase detail block: always from MD (rich content for planner LLM)
     echo ""
     echo "--- Phase ${PHASE_NUM} Detail ---"
     awk -v pnum="$PHASE_NUM" '
       /^### Phase / {
         line = $0
         sub(/^### Phase /, "", line)
         sub(/:.*/, "", line)
         gsub(/ .*/, "", line)
         if (line == pnum) { found=1; print; next }
         else if (found) { exit }
       }
       found { print }
     ' "$ROADMAP_FILE" 2>/dev/null
   fi
   ```
